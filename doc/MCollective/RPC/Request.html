def initialize(msg, ddl)
  @time = msg[:msgtime]
  @action = msg[:body][:action]
  @data = msg[:body][:data]
  @sender = msg[:senderid]
  @agent = msg[:body][:agent]
  @uniqid = msg[:requestid]
  @caller = msg[:callerid] || "unknown"
  @ddl = ddl
end

def should_respond?
  return @data[:process_results] if @data.include?(:process_results)

  return true
end
