def to_hash
  stats = {:validated => @validated,
    :unvalidated => @unvalidated,
    :passed => @passed,
    :filtered => @filtered,
    :starttime => @starttime,
    :total => @total,
    :ttlexpired => @ttlexpired,
    :replies => @replies}

  reply = {:stats => stats,
    :threads => [],
    :pid => Process.pid,
    :times => {} }

  ::Process.times.each_pair{|k,v|
    k = k.to_sym
    reply[:times][k] = v
  }

  Thread.list.each do |t|
    reply[:threads] << "#{t.inspect}"
  end

  reply[:agents] = Agents.agentlist
  reply
end
