def gem_command(host)
  if host['platform'] =~ /windows/
    "env PATH=\"#{host['privatebindir']}:${PATH}\" cmd /c gem"
  else
    "env PATH=\"#{host['privatebindir']}:${PATH}\" gem"
  end
end
