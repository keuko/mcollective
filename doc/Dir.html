 72 def self.mktmpdir(prefix_suffix=nil, tmpdir=nil)
 73   case prefix_suffix
 74   when nil

 76     suffix = ""

 80   when Array

 85   end
 86   tmpdir ||= Dir.tmpdir
 87   t = Time.now.strftime("%Y%m%d")
 88   n = nil
 89   begin
 90     path = "#{tmpdir}/#{prefix}#{t}-#{$$}-#{rand(0x100000000).to_s(36)}"
 91     path << "-#{n}" if n
 92     path << suffix
 93     Dir.mkdir(path, 0700)
